import { db, collection, addDoc } from './firebase-config.js'; const form = document.getElementById('registroForm'); if (form) { form.addEventListener('submit', async (e) => { e.preventDefault(); const submitBtn = form.querySelector('button'); submitBtn.innerText = "PROCESANDO..."; submitBtn.disabled = true; try { // Capturamos los campos de texto de forma precisa const nombre = form.querySelector('input[placeholder*="Ej: Juan"]').value; const cedula = form.querySelector('input[placeholder*="8 dígitos"]').value; const vehiculo = form.querySelector('input[placeholder*="Nissan"]').value; const placas = form.querySelector('input[placeholder*="UTP"]').value; // Guardar en Firestore await addDoc(collection(db, "solicitudes_tecnicos"), { nombre: nombre, cedula: cedula || "No provista", vehiculo: vehiculo, placas: placas, estatus: "pendiente", fechaRegistro: new Date().toISOString() }); alert("¡ÉXITO! Tu solicitud ha sido enviada a FixGo."); form.reset(); } catch (error) { console.error("Error detallado:", error); alert("Error al enviar: " + error.message); } finally { submitBtn.innerText = "ENVIAR SOLICITUD DE ALTA"; submitBtn.disabled = false; } }); }
