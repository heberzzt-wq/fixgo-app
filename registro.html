import { db, collection, addDoc } from './firebase-config.js'; const form = document.getElementById('registroForm'); if (form) { form.addEventListener('submit', async (e) => { e.preventDefault(); const submitBtn = form.querySelector('button'); submitBtn.innerText = "PROCESANDO..."; submitBtn.disabled = true; try { // Seleccionamos todos los campos de texto del formulario const campos = form.querySelectorAll('input[type="text"]'); // Los capturamos por su posición (orden en el que aparecen en el HTML) const datos = { nombre: campos[0] ? campos[0].value : "Sin nombre", cedula: campos[1] ? campos[1].value : "Sin cédula", vehiculo: campos[2] ? campos[2].value : "Sin vehículo", placas: campos[3] ? campos[3].value : "Sin placas", estatus: "pendiente", fechaRegistro: new Date().toISOString() }; console.log("Intentando guardar datos:", datos); // Guardar en la colección de Firebase await addDoc(collection(db, "solicitudes_tecnicos"), datos); alert("¡ÉXITO! Tu solicitud ha sido enviada correctamente."); form.reset(); } catch (error) { console.error("Error completo:", error); alert("Error al enviar. Verifica tu conexión a Internet."); } finally { submitBtn.innerText = "ENVIAR SOLICITUD DE ALTA"; submitBtn.disabled = false; } }); }
